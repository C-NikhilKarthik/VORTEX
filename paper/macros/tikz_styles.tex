\newcommand{\databar}[3]{%
  \tikz[baseline=-0.6ex]{
    \fill[#3, draw=#3!80!black, line width=0.5pt, rounded corners=1pt] (0,-0.1) rectangle (#1, 0.2);
    \node[anchor=west, font=\scriptsize\sffamily, inner sep=2pt] at (#1, 0.05) {#2};
  }%
}
